
// qwe has been designed to:
//   - attach complex behaviours to any JS object.
//   - 'do not disturb policy': a user object shell not be touched in any way except by adding the attribute '__behaviour__'

// Now let's suppose that you want to transform your object in a powerful and robust state machine

// Here is how you do it
var qwe = require('qwe');       // Load the library
var Top = qwe.stCreate('Top');  // Create a State
qwe.smInit(this, Top);          // Initialize the State Machine

// Congratulations you have created your Hierarchical state Machine !
